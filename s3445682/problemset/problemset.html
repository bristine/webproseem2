def front_back(a, b):
	if len(a)%2 == 0:
		numa = len(a)/2
		a1 = a[:numa]
		a2 = a[numa:]
	else:
		numa = len(a)/2 + 1
		a1 = a[:numa]
		a2 = a[numa:]
	if len(b)%2 == 0:
		numb = len(b)/2
		b1 = b[:numb]
		b2 = b[numb:]
	else:
		numb = len(b)/2 + 1
		b1 = b[:numb]
		b2 = b[numb:]
	new = a1 + b1 + a2 + b2
	return new

def sort_last(tuples):
	for n in range(0,len(tuples)):
		for a in range(0,len(tuples)-1):
			take_a = tuples[a]
			take_a1 = tuples[a+1]
			if take_a[-1] > take_a1[-1]:
				t = tuples[a]
				tuples[a] = tuples[a+1]
				tuples[a+1] = t
	return tuples

def remove_adjacent(nums):
	if nums == []:
		return nums
	else:
		nums.sort()
		newList = [nums[0]]
		for a in range (1, len(nums)):
			x = nums[a]
			if nums[a] != nums[a-1]:
				newList.append(x)
		return newList
